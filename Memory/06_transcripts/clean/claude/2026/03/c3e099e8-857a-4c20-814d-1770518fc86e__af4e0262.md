import Database from 'better-sqlite3'
import { mkdirSync, existsSync } from 'node:fs'
import { dirname } from 'node:path'
import type { ManifestEntry } from '@openmnemo/types'

const PK_COLUMNS = ['client','project','session_id','raw_sha256'] as const
const DATA_COLUMNS = [
  'title','started_at','imported_at','cwd','branch',
  'raw_source_path','raw_upload_permission',
  'global_raw_path','global_clean_path',
  'repo_raw_path','repo_clean_path','repo_manifest_path',
  'message_count','tool_event_count',
  'content','commit_layer',
] as const
const ALL_COLUMNS = [...PK_COLUMNS, ...DATA_COLUMNS] as const

const CREATE_TABLE_SQL = `
  CREATE TABLE IF NOT EXISTS transcripts (
    client TEXT NOT NULL, project TEXT NOT NULL,
    session_id TEXT NOT NULL, raw_sha256 TEXT NOT NULL,
    title TEXT NOT NULL, started_at TEXT NOT NULL, imported_at TEXT NOT NULL,
    cwd TEXT NOT NULL, branch TEXT NOT NULL,
    raw_source_path TEXT NOT NULL, raw_upload_permission TEXT NOT NULL,
    global_raw_path TEXT NOT NULL, global_clean_path TEXT NOT NULL,
    repo_raw_path TEXT NOT NULL, repo_clean_path TEXT NOT NULL,
    repo_manifest_path TEXT NOT NULL,
    message_count INTEGER NOT NULL, tool_event_count INTEGER NOT NULL,
    content TEXT NOT NULL DEFAULT '',
    commit_layer TEXT NOT NULL DEFAULT '',
    PRIMARY KEY (client, project, session_id, raw_sha256)
  )
`

const UPSERT_SQL = `
  INSERT INTO transcripts (${ALL_COLUMNS.join(', ')})
  VALUES (${ALL_COLUMNS.map(() => '?').join(', ')})
  ON CONFLICT(${PK_COLUMNS.join(', ')}) DO UPDATE SET
    ${DATA_COLUMNS.map((c) => `${c} = excluded.${c}`).join(',\n    ')}
`

const CREATE_FTS_SQL = `
  CREATE VIRTUAL TABLE IF NOT EXISTS transcripts_fts USING fts4(
    content='transcripts',
    title, cwd, branch, content, commit_layer
  )
`

const REBUILD_FTS_SQL = `INSERT INTO transcripts_fts(transcripts_fts) VALUES('rebuild')`

const SEARCH_SQL = `
  SELECT t.client, t.project, t.session_id, t.title, t.cwd, t.branch, t.started_at
  FROM transcripts_fts
  JOIN transcripts t ON transcripts_fts.docid = t.rowid
  WHERE transcripts_fts MATCH ?
  ORDER BY t.started_at DESC
  LIMIT ?
`

export interface SearchResult { client: string; project: string; session_id: string; title: string; cwd: string; branch: string; started_at: string }

export function sanitizeFtsQuery(raw: string): string {
  return raw.replace(/[^\p{L}\p{N}\s]/gu, ' ').trim().replace(/\s+/g, ' ')
}

function openDb(dbPath: string): InstanceType<typeof Database> {
  mkdirSync(dirname(dbPath), { recursive: true })
  const db = new Database(dbPath)
  db.pragma('journal_mode = WAL')
  db.pragma('busy_timeout = 5000')
  return db
}

function migrateSchema(db: InstanceType<typeof Database>): void {
  for (const col of ['content', 'commit_layer']) {
    try {
      db.exec(`ALTER TABLE transcripts ADD COLUMN ${col} TEXT NOT NULL DEFAULT ''`)
    } catch (e: unknown) {
      if (!(e instanceof Error) || !e.message.includes('duplicate column name')) throw e
    }
  }
}

function migrateFts(db: InstanceType<typeof Database>): void {
  const row = db.prepare(
    "SELECT sql FROM sqlite_master WHERE type='table' AND name='transcripts_fts'"
  ).get() as { sql: string } | undefined
  if (!row || !row.sql.includes('commit_layer')) {
    db.exec('DROP TABLE IF EXISTS transcripts_fts')
    db.exec(CREATE_FTS_SQL)
    db.exec(REBUILD_FTS_SQL)
  }
}

export function initSchema(dbPath: string): void {
  const db = openDb(dbPath)
  try {
    db.exec(CREATE_TABLE_SQL)
    migrateSchema(db)
    db.exec(CREATE_FTS_SQL)
    migrateFts(db)
  } finally { db.close() }
}

export function rebuildFtsIndex(dbPath: string): void {
  const db = openDb(dbPath)
  try {
    db.exec(CREATE_FTS_SQL)
    db.exec(REBUILD_FTS_SQL)
  } finally { db.close() }
}

export function searchTranscripts(dbPath: string, query: string, limit = 20): SearchResult[] {
  const sanitized = sanitizeFtsQuery(query)
  if (!sanitized) return []
  if (!existsSync(dbPath)) return []
  const db = new Database(dbPath, { readonly: true })
  try {
    return db.prepare(SEARCH_SQL).all(sanitized, limit) as SearchResult[]
  } catch { return [] } finally { db.close() }
}

export function searchTranscriptsByColumns(dbPath: string, query: string, columns: string[], limit = 20): SearchResult[] {
  const sanitized = sanitizeFtsQuery(query)
  if (!sanitized || columns.length === 0) return []
  if (!existsSync(dbPath)) return []
  const columnQuery = columns.map((col) => `${col}:${sanitized}`).join(' OR ')
  const db = new Database(dbPath, { readonly: true })
  try {
    return db.prepare(SEARCH_SQL).all(columnQuery, limit) as SearchResult[]
  } catch { return [] } finally { db.close() }
}

export function upsertSearchIndex(dbPath: string, manifest: ManifestEntry): void {
  const db = openDb(dbPath)
  try {
    db.exec(CREATE_TABLE_SQL)
    migrateSchema(db)
    db.exec(CREATE_FTS_SQL)
    migrateFts(db)
    const record: Record<string, unknown> = { ...manifest }
    const params: (string | number)[] = ALL_COLUMNS.map((col) => {
      const value = record[col]
      if (value === undefined || value === null) return ''
      return typeof value === 'number' ? value : String(value)
    })
    db.prepare(UPSERT_SQL).run(params)
    db.exec(REBUILD_FTS_SQL)
  } finally { db.close() }
}

**1. CRITICAL: Silent error swallowing in searchTranscripts and searchTranscriptsByColumns**

- Issue: Catches all exceptions including database corruption, permission errors, and query syntax errors. Caller has no way to distinguish "no results" from "database error". This masks bugs in FTS query construction.
- Impact: Bugs in `searchTranscriptsByColumns` column query formatting (line 77) could silently fail without detection.

**2. HIGH: Unsafe FTS query construction in searchTranscriptsByColumns**
- Line 77: `const columnQuery = columns.map((col) => \`${col}:${sanitized}\`).join(' OR ')`
- Issue: Column names are not validated. If `columns` array contains user input or untrusted values, this could inject FTS operators or malformed syntax. The `sanitized` query is safe, but column names bypass validation.
- Impact: Potential FTS query injection if columns come from user input.
- Fix: Validate/whitelist column names against known safe set: `['title', 'cwd', 'branch', 'content', 'commit_layer']`

**3. HIGH: Race condition in migrateFts**
- Lines 60-66: Checks if FTS table exists, then drops and recreates it
- Issue: Between the check and the drop, another process could modify the table. Also, `REBUILD_FTS_SQL` is called after every upsert (line 95), which is expensive and unnecessary—FTS4 with `content=` option auto-syncs.
- Impact: Unnecessary performance degradation; potential data loss if concurrent access occurs.
- Fix: Use `IF NOT EXISTS` in CREATE, remove `REBUILD_FTS_SQL` from upsertSearchIndex, or use FTS5 with auto-rebuild.

**4. MEDIUM: Missing global_manifest_path in ALL_COLUMNS**
- Line 10: `ManifestEntry` includes `global_manifest_path`, but it's not in `DATA_COLUMNS` or `ALL_COLUMNS`
- Issue: This field is silently dropped during upsert. If it's needed later, data loss occurs.
- Impact: Incomplete manifest persistence.
- Fix: Add `'global_manifest_path'` to `DATA_COLUMNS` and update CREATE_TABLE_SQL.

**5. MEDIUM: Inconsistent null handling in upsertSearchIndex**
- Lines 88-92: Converts `undefined` and `null` to empty string `''`
- Issue: This loses information—can't distinguish "field was null" from "field was empty string". For optional fields like `content` and `commit_layer`, this may be intentional, but for required fields like `title`, it masks data quality issues.
- Impact: Silent data corruption if required fields are missing.
- Fix: Validate that required fields are present before upsert; throw error if missing.

**6. MEDIUM: FTS4 content table sync issue**
- Line 95: `db.exec(REBUILD_FTS_SQL)` after every upsert
- Issue: FTS4 with `content=` option requires manual rebuild only if the content table is modified outside FTS. Since we're using prepared statements, this is unnecessary overhead.
- Impact: Performance: every upsert triggers a full FTS rebuild.
- Fix: Remove the rebuild call; FTS4 auto-syncs with content table.

**7. LOW: Missing index on search results**
- Line 50: SEARCH_SQL joins on `rowid` but no explicit index
- Issue: FTS4 provides implicit rowid mapping, but performance could degrade with large tables.
- Impact: Slow searches on large datasets.
- Fix: Consider adding explicit index or migrating to FTS5.
