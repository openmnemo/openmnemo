### 2. **Report Serve 命令实现**

**文件**: `D:/demo1/openmnemo/packages/cli/src/cmd-report.ts`

**HTTP Server 框架**: 使用 Node.js 原生 `node:http` 模块（无外部依赖）

**核心实现**:
```typescript
// 创建 HTTP 服务器
export function createReportServer(opts: CmdReportServeOptions): import('node:http').Server {
  const root = resolve(opts.dir)
  const port = opts.port
  
  const server = createServer((req, res) => {
    // 1. 解码 URL 路径
    let urlPath = decodeURIComponent((req.url ?? '/').split('?')[0] ?? '/')
    
    // 2. 路径安全检查（防止目录遍历）
    if (!filePath.startsWith(root + sep) && filePath !== root) {
      res.writeHead(403, { 'Content-Type': 'text/plain' })
      res.end('Forbidden')
      return
    }
    
    // 3. 目录请求自动转向 index.html
    if (existsSync(filePath) && statSync(filePath).isDirectory()) {
      filePath = join(filePath, 'index.html')
    }
    
    // 4. 流式传输文件（支持 MIME 类型）
    const contentType = MIME[ext] ?? 'application/octet-stream'
    res.writeHead(200, { 'Content-Type': contentType })
    const stream = createReadStream(filePath)
    stream.pipe(res)
  })
  
  server.listen(port, () => {
    console.log(`Serving ${root}`)
    console.log(`Open: http://localhost:${port}/`)
  })
}
```

**MIME 类型支持**:
- `.html` → `text/html; charset=utf-8`
- `.css` → `text/css`
- `.js` → `application/javascript`
- `.json` → `application/json`
- `.svg` → `image/svg+xml`
- `.ico` → `image/x-icon`
