function migrateSchema(db): void {
  for (const col of ['content', 'commit_layer']) {
    try {
      db.exec(`ALTER TABLE transcripts ADD COLUMN ${col} TEXT NOT NULL DEFAULT ''`)
    } catch (e) {
      if (!(e instanceof Error) || !e.message.includes('duplicate column name')) throw e
    }
  }
}

// migrateFts: drops+recreates FTS table if commit_layer column missing
function migrateFts(db): void {
  const row = db.prepare(
    "SELECT sql FROM sqlite_master WHERE type='table' AND name='transcripts_fts'"
  ).get() as { sql: string } | undefined
  if (!row || !row.sql.includes('commit_layer')) {
    db.exec('DROP TABLE IF EXISTS transcripts_fts')
    db.exec(CREATE_FTS_SQL)
    db.exec(REBUILD_FTS_SQL)
  }
}

// searchTranscriptsByColumns: column-scoped FTS4 MATCH
export function searchTranscriptsByColumns(dbPath, query, columns, limit = 20): SearchResult[] {
  const sanitized = sanitizeFtsQuery(query)
  if (!sanitized || columns.length === 0) return []
  if (!existsSync(dbPath)) return []
  const columnQuery = columns.map((col) => `${col}:${sanitized}`).join(' OR ')
  const db = new Database(dbPath, { readonly: true })
  try {
    return db.prepare(SEARCH_SQL).all(columnQuery, limit) as SearchResult[]
  } catch { return [] } finally { db.close() }
}

### 1. **FTS4 Column Filter Syntax — CRITICAL**
**File:** `packages/core/src/transcript/db.ts` (line 253)  
**Issue:** The column filter syntax `col:term OR col2:term` is **not valid FTS4 syntax**.

FTS4 column filters use the format `column:term` but the `OR` operator at the FTS level doesn't work as intended. The correct syntax should be `(commit_layer:${sanitized})` as a single MATCH clause, not `commit_layer:term OR content:term`.

**Impact:** `searchTranscriptsByColumns(['commit_layer', 'content'], query)` will likely fail silently or return incorrect results because the MATCH clause becomes malformed.

**Severity:** CRITICAL — Search layer 1 and 3 in `searchRecall` may not work correctly.

### 2. **Race Condition in Migration — HIGH**
**File:** `packages/core/src/transcript/db.ts` (lines 156-164, 170-180)  
**Issue:** `migrateSchema` and `migrateFts` are called inside `upsertSearchIndex` without transaction protection. Multiple concurrent writers can:
- Both attempt `ALTER TABLE ADD COLUMN` simultaneously
- One process drops FTS table while another is querying it
- Partial migration state leaves DB inconsistent

**Current mitigation:** `busy_timeout = 5000ms` only delays, doesn't prevent race conditions.

**Severity:** HIGH — Under concurrent load (multiple sync processes), DB corruption or query failures are possible.

### 3. **Content Field Size — Performance Risk — HIGH**
**File:** `packages/core/src/transcript/import.ts` (line 106-107)  
**Issue:** Clean markdown files can be 100KB+ (full session transcripts). Storing entire content in SQLite TEXT column and rebuilding FTS index on every upsert is expensive:
- FTS rebuild is O(n) for all rows
- Large TEXT fields slow down FTS queries
- No pagination or chunking strategy

**Current behavior:** Every `upsertSearchIndex` call triggers `REBUILD_FTS_SQL` (line 287 in db.ts), even for single-row inserts.

**Severity:** HIGH — Performance degrades with large transcripts; FTS rebuild becomes bottleneck.
